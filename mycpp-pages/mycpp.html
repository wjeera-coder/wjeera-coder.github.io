  • mt random number generator 

    
  #include <iostream>
  #include <random>

  using namespace std;
  mt19937 mt_rand(time(0));
  double myrand(){
    return (1.0*mt_rand()/mt_rand.max());
  }
  int main(){
    for(int i=1;i<=20;i++){
      for(int j=1;j<=5;j++){
        printf("%f ",myrand());
      }
      printf("\n");
    }
    return 0;
  }
