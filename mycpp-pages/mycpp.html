My C++ Page

  • Using GNU g++ 

    
  //hello.c
  #include <iostream>
  int main (){
    std::cout << "Hello, world!" << std::endl;
    return 0;
  }
  
    
  //hello.cc
  #include <iostream>
  #using namespace std;
  int main (){
    cout << "Hello, world!" << std::endl;
    return 0;
  }
    The program can be compiled with the following command line:
    $ gcc -Wall hello.c -o hello

    or
    $ g++ -Wall hello.cc -o hello

  • mt random number generator 

    
  //myrand.c
  #include <stdio.h>
  #include <random>

  mt19937 mt_rand(time(0));
  double myrand(){
    return (1.0*mt_rand()/mt_rand.max());
  }
  int main(){
    for(int i=1;i<=20;i++){
      for(int j=1;j<=5;j++){
        printf("%f ",myrand());
      }
      printf("\n");
    }
    return 0;
  }
    The program can be compiled with the following command line:
    $ gcc -Wall myrand.c -o myrand
